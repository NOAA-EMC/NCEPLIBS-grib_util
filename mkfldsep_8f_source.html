5 C> Generates a TOC Flag Field Separator Block used to separate
6 C> WMO Bulletins within a transmission file to be ingested in TOC's
7 C> FTP Input Service, which can be used to disseminate WMO buletins.
8 C> ( see http://weather.gov/tg/ftpingest.html )
9 C>
10 C> This routine can generate different flag field separator blocks
11 C> depending on the value of variable iopt.
12 C>
13 C> Bulletin "Flag Field Separator" block - OPTION 1 (old)
14 C> bytes 1 - 4 marker string (####)
15 C> 5 - 7 block length [018 fixed value]
16 C> 8 - 13 total length of bulletin in bytes [octets]
17 C> (not including the flag field block)
18 C> 14 - 17 marker string (####)
19 C> 18 line Feed (ASCII "0A")
20 C>
21 C> Bulletin "Flag Field Separator" block - OPTION 1a (new)
22 C> bytes 1 - 4 marker string (####)
23 C> 5 - 7 block length (nnn) - value always greater than 018
24 C> 8 - 18 total length of bulletin in bytes [octets]
25 C> (not including the flag field block)
26 C> 19 - nnn-5 reserved for future use
27 C> nnn-4 - nnn-1 marker string (####)
28 C> nnn line Feed (ASCII "0A")
29 C>
30 C> Bulletin "Flag Field Separator" block - OPTION 2 (limited)
31 C> bytes 1 - 4 marker string (****)
32 C> 5 - 14 total length of bulletin in bytes [octets]
33 C> (not including the flag field block)
34 C> 15 - 18 marker string (****)
35 C> 19 line Feed (ASCII "0A")
36 C>
37 C>
38 C> PROGRAM HISTORY LOG:
39 C> 2002-09-16 Gilbert ORIGINAL AUTHOR
40 C>
41 C> USAGE: call mkfldsep(csep,iopt,lenin,lenbull,lenout)
42 C> @param[out] csep*(*) Character array containing the flag field separator.
43 C> @param[in] iopt Flag Field Separator block option:
44 C> = 1: Separator block for use with alphanumeric bulletins.
45 C> if lenin <= 18 and lenbull <= 999999,
46 C> OPTION 1 block will be generated.
47 C> if lenin > 18 or lenbull > 999999,

51 C> ignored, if iopt=2.

54 C> @param[out] lenout Integer length of the flag field separator block.
55 C>
56  subroutine mkfldsep(csep,iopt,lenin,lenbull,lenout)
57 C
58  character*(*),intent(out) :: csep
59  integer,intent(in) :: iopt,lenin,lenbull
60  integer,intent(out) :: lenout
61 C
62  character(len=4),parameter :: cstar='****',clb='####'
63 C
64  if (iopt.eq.1) then
65  if ( lenin .le. 18 .and. lenbull .le. 999999 ) then
66  ! Create OPTION 1 separator block
67  csep(1:4)=clb
68  csep(5:7)='018'
69  write(csep(8:13),fmt='(I6.6)') lenbull
70  csep(14:17)=clb
71  csep(18:18)=char(10)
72  lenout=18
73  else ! Create OPTION 1a separator block
74  nnn=lenin
75  if ( nnn.lt.23 ) nnn=23
76  csep(1:4)=clb
77  write(csep(5:7),fmt='(I3.3)') nnn
78  write(csep(8:18),fmt='(I11.11)') lenbull
79  csep(19:nnn-5)='0'
80  csep(nnn-4:nnn-1)=clb
81  csep(nnn:nnn)=char(10)
82  lenout=nnn
83  endif
84  elseif (iopt.eq.2) then ! Create OPTION 2 separator block
85  csep(1:4)=cstar
86  write(csep(5:14),fmt='(I10.10)') lenbull
87  csep(15:18)=cstar
88  csep(19:19)=char(10)
89  lenout=19
90  else
91  print *,"mkfldsep: Option ",iopt," not recognized."
92  csep(1:lenin)=' '
93  endif
94 C
95  return
96  end
