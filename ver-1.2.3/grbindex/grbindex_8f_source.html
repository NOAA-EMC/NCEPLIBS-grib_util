2 C> WRITE AN INDEX FILE
3 C> @author IREDELL @date 1998-09-15
4 C
5 C> PROGRAM CREATES AN INDEX FILE FROM A GRIB FILE.
6 C> THE INDEX FILE SERVES AS A TABLE OF CONTENTS FOR THE GRIB FILE,
7 C> ENABLING QUICK ACCESS TO THE DATA. THE GRIB FILE MUST BE UNBLOCKED,
8 C> BUT THERE CAN BE A GAP BEFORE THE FIRST GRIB MESSAGE OF AT MOST
9 C> 32000 BYTES AND GAPS BETWEEN MESSAGES OF AT MOST 4000 BYTES.
10 C> THE TWO FILE NAMES ARE RETRIEVED FROM THE COMMAND LINE ARGUMENTS.
11 C> THE FIRST ARGUMENT IS THE NAME OF THE INPUT GRIB FILE.
12 C> THE SECOND ARGUMENT IS THE NAME OF THE OUTPUT INDEX FILE.
13 C> CURRENTLY, ONLY VERSION 1 OF GRIB CAN BE READ.
14 C> VERSION 1 OF THE INDEX FILE HAS THE FOLLOWING FORMAT:
15 C> 81-BYTE S.LORD HEADER WITH 'GB1IX1' IN COLUMNS 42-47 FOLLOWED BY
16 C> 81-BYTE HEADER WITH NUMBER OF BYTES TO SKIP BEFORE INDEX RECORDS,
17 C> NUMBER OF BYTES IN EACH INDEX RECORD, NUMBER OF INDEX RECORDS,
18 C> AND GRIB FILE BASENAME WRITTEN IN FORMAT ('IX1FORM:',3I10,2X,A40).
19 C> EACH FOLLOWING INDEX RECORD CORRESPONDS TO A GRIB MESSAGE
20 C> AND HAS THE INTERNAL FORMAT:
21 C> - BYTE 001-004: BYTES TO SKIP IN DATA FILE BEFORE GRIB MESSAGE
22 C> - BYTE 005-008: BYTES TO SKIP IN MESSAGE BEFORE PDS
23 C> - BYTE 009-012: BYTES TO SKIP IN MESSAGE BEFORE GDS (0 IF NO GDS)
24 C> - BYTE 013-016: BYTES TO SKIP IN MESSAGE BEFORE BMS (0 IF NO BMS)
25 C> - BYTE 017-020: BYTES TO SKIP IN MESSAGE BEFORE BDS
26 C> - BYTE 021-024: BYTES TOTAL IN THE MESSAGE
27 C> - BYTE 025-025: GRIB VERSION NUMBER
28 C> - BYTE 026-053: PRODUCT DEFINITION SECTION (PDS)
29 C> - BYTE 054-095: GRID DEFINITION SECTION (GDS) (OR NULLS)
30 C> - BYTE 096-101: FIRST PART OF THE BIT MAP SECTION (BMS) (OR NULLS)
31 C> - BYTE 102-112: FIRST PART OF THE BINARY DATA SECTION (BDS)
32 C> - BYTE 113-172: (OPTIONAL) BYTES 41-100 OF THE PDS
33 C> - BYTE 173-184: (OPTIONAL) BYTES 29-40 OF THE PDS
34 C> - BYTE 185-320: (OPTIONAL) BYTES 43-178 OF THE GDS
35 C>
36 C> PROGRAM HISTORY LOG:
37 C> - 92-11-22 IREDELL
38 C> - 94-06-08 EBISUZAKI - ELIMINATE ISHELL CALLS
39 C> - 94-08-26 IREDELL - 40 BYTE PDS EXTENSION
40 C> - 95-10-31 IREDELL - CONSIDERABLY REDUCE I/O
41 C> - 96-10-31 IREDELL - AUGMENTED OPTIONAL DEFINITIONS TO BYTE 320
42 C> - 1999-04-27 GILBERT - Changed CALL EXIT(N) to CALL ERREXIT(N) so that
43 C> the proper error return value is passed back to
44 C> the shell.
45 C> - 2012-07-31 VUONG - CHANGED HOSTNAME TO HOSTNAM
46 C>
47 C> USAGE: grbindex gribfile indexfile
48 C>
49 C> INPUT FILE:
50 C> - gribfile UNBLOCKED GRIB FILE
51 C>
52 C> OUTPUT FILE:
53 C> - indexfile UNBLOCKED INDEX FILE
54 C>
55 C> SUBPROGRAMS CALLED:
56 C> - IARGC COUNT THE COMMAND LINE ARGUMENTS
57 C> - GETARG GET COMMAND LINE ARGUMENT
58 C> - WRGI1H WRITE INDEX HEADERS
59 C> - GETGIR GET INDEX BUFFER
60 C> - BAWRITE BYTE-ADDRESSABLE WRITE
61 C> - ERRMSG WRITE A MESSAGE TO STDERR
62 C> - ERREXIT EXIT WITH RETURN CODE
63 C>
64 C> EXIT STATES:
65 C> - COND = 0 - SUCCESSFUL RUN
66 C> - COND = 1 - GRIB MESSAGE NOT FOUND
67 C> - COND = 2 - INCORRECT ARGUMENTS
68 C> - COND = 8 - ERROR ACCESSING FILE
69 C>
70  PROGRAM grbindex
71  parameter(msk1=32000,msk2=4000)
72  CHARACTER cgb*256,cgi*256
73  parameter(mbuf=256*1024)
74  CHARACTER cbuf(mbuf)
75  CHARACTER carg*300
76  INTEGER narg,iargc
77 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
78 C GET ARGUMENTS
79  narg=iargc()
80  IF(narg.NE.2) THEN
81  CALL errmsg('grbindex: Incorrect usage')
82  CALL errmsg('Usage: grbindex gribfile indexfile')
83  CALL errexit(2)
84  ENDIF
85  CALL getarg(1,cgb)
86  ncgb=len_trim(cgb)
87  CALL baopenr(11,cgb(1:ncgb),ios)
88  CALL baseto(1,1)
89  IF(ios.NE.0) THEN
90  lcarg=len('grbindex: Error accessing file '//cgb(1:ncgb))
91  carg(1:lcarg)='grbindex: Error accessing file '//cgb(1:ncgb)
92  CALL errmsg(carg(1:lcarg))
93  CALL errexit(8)
94  ENDIF
95  CALL getarg(2,cgi)
96  ncgi=len_trim(cgi)
97  CALL baopen(31,cgi(1:ncgi),ios)
98  IF(ios.NE.0) THEN
99  lcarg=len('grbindex: Error accessing file '//cgi(1:ncgi))
100  carg(1:lcarg)='grbindex: Error accessing file '//cgi(1:ncgi)
101  CALL errmsg(carg(1:lcarg))
102  CALL errexit(8)
103  ENDIF
104 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
105 C WRITE INDEX FILE
106  mnum=0
107  CALL getgir(11,msk1,msk2,mnum,mbuf,cbuf,nlen,nnum,irgi)
108  IF(irgi.GT.1.OR.nnum.EQ.0) THEN
109  CALL errmsg('grbindex: No GRIB messages detected in file '
110  & //cgb(1:ncgb))
111  CALL baclose(11,iret)
112  CALL baclose(31,iret)
113  CALL errexit(1)
114  ENDIF
115  mnum=mnum+nnum
116  CALL wrgi1h(31,nlen,mnum,cgb(1:ncgb))
117  iw=162
118  CALL bawrite(31,iw,nlen*nnum,kw,cbuf)
119  iw=iw+nlen*nnum
120 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
121 C EXTEND INDEX FILE IF INDEX BUFFER LENGTH GREATER THAN MBUF
122  IF(irgi.EQ.1) THEN
123  dowhile(irgi.EQ.1.AND.nnum.GT.0)
124  CALL getgir(11,msk1,msk2,mnum,mbuf,cbuf,nlen,nnum,irgi)
125  IF(irgi.LE.1.AND.nnum.GT.0) THEN
126  mnum=mnum+nnum
127  CALL bawrite(31,iw,nlen*nnum,kw,cbuf)
128  iw=iw+nlen*nnum
129  ENDIF
130  ENDDO
131  CALL wrgi1h(31,nlen,mnum,cgb(1:ncgb))
132  ENDIF
133  CALL baclose(11,iret)
134  CALL baclose(31,iret)
135 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
136  END
