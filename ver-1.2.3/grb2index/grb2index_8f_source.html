2 C> WRITE A GRIB2 INDEX FILE
3 C> @author GILBERT @date 2002-01-03
4 C
5 C> PROGRAM CREATES AN INDEX FILE FROM A GRIB2 FILE.
6 C> THE INDEX FILE SERVES AS A TABLE OF CONTENTS FOR THE GRIB FILE,
7 C> ENABLING QUICK ACCESS TO THE DATA. THE GRIB FILE MUST BE UNBLOCKED,
8 C> BUT THERE CAN BE A GAP BEFORE THE FIRST GRIB MESSAGE OF AT MOST
9 C> 32000 BYTES AND GAPS BETWEEN MESSAGES OF AT MOST 4000 BYTES.
10 C> THE TWO FILE NAMES ARE RETRIEVED FROM THE COMMAND LINE ARGUMENTS.
11 C> THE FIRST ARGUMENT IS THE NAME OF THE INPUT GRIB FILE.
12 C> THE SECOND ARGUMENT IS THE NAME OF THE OUTPUT INDEX FILE.
13 C> FOR THIS PROGRAM, ONLY GRIB VERSION 2 CAN BE READ.
14 C> VERSION 1 OF THE INDEX FILE HAS THE FOLLOWING FORMAT:
15 C> 81-BYTE S.LORD HEADER WITH 'GB2IX1' IN COLUMNS 42-47 FOLLOWED BY
16 C> 81-BYTE HEADER WITH NUMBER OF BYTES TO SKIP BEFORE INDEX RECORDS,
17 C> TOTAL LENGTH IN BYTES OF THE INDEX RECORDS, NUMBER OF INDEX RECORDS,
18 C> AND GRIB FILE BASENAME WRITTEN IN FORMAT ('IX1FORM:',3I10,2X,A40).
19 C> EACH FOLLOWING INDEX RECORD CORRESPONDS TO A FIELD IN A GRIB2 MESSAGE
20 C> AND HAS THE INTERNAL FORMAT:
21 C> - BYTE 001 - 004: LENGTH OF INDEX RECORD
22 C> - BYTE 005 - 008: BYTES TO SKIP IN DATA FILE BEFORE GRIB MESSAGE
23 C> - BYTE 009 - 012: BYTES TO SKIP IN MESSAGE BEFORE GDS
24 C> - BYTE 013 - 016: BYTES TO SKIP IN MESSAGE BEFORE PDS
25 C> - BYTE 017 - 020: BYTES TO SKIP IN MESSAGE BEFORE DRS
26 C> - BYTE 021 - 024: BYTES TO SKIP IN MESSAGE BEFORE BMS
27 C> - BYTE 025 - 032: BYTES TOTAL IN THE MESSAGE
28 C> - BYTE 033 - 033: GRIB VERSION NUMBER ( CURRENTLY 2 )
29 C> - BYTE 034 - 034: MESSAGE DISCIPLINE
30 C> - BYTE 035 - 036: FIELD NUMBER WITHIN GRIB2 MESSAGE
31 C> - BYTE 037 - II: IDENTIFICATION SECTION (IDS)
32 C> - BYTE II+1- JJ: GRID DEFINITION SECTION (GDS)
33 C> - BYTE JJ+1- KK: PRODUCT DEFINITION SECTION (PDS)
34 C> - BYTE KK+1- LL: THE DATA REPRESENTATION SECTION (DRS)
35 C> - BYTE LL+1-LL+6: FIRST 6 BYTES OF THE BIT MAP SECTION (BMS)
36 C>
37 C> PROGRAM HISTORY LOG:
38 C> - 92-11-22 IREDELL
39 C> - 2002-01-03 GILBERT - MODIFIED FROM PROGRAM GRBINDEX TO WORK WITH GRIB2
40 C> - 2005-02-25 GILBERT - Removed buffering option (see baseto).
41 C> - 2012-08-01 VUONG - CHANGED HOSTNAME TO HOSTNAM
42 C>
43 C> USAGE: grb2index gribfile indexfile
44 C>
45 C> INPUT FILE:
46 C> - gribfile GRIB2 FILE
47 C>
48 C> OUTPUT FILE:
49 C> - indexfile UNBLOCKED INDEX FILE
50 C>
51 C> SUBPROGRAMS CALLED:
52 C> - IARGC COUNT THE COMMAND LINE ARGUMENTS
53 C> - GETARG GET COMMAND LINE ARGUMENT
54 C> - WRGI1H WRITE INDEX HEADERS
55 C> - GETG2IR GET INDEX BUFFER
56 C> - BAWRITE BYTE-ADDRESSABLE WRITE
57 C> - ERRMSG WRITE A MESSAGE TO STDERR
58 C> - ERREXIT EXIT WITH RETURN CODE
59 C>
60 C> EXIT STATES:
61 C> - COND = 0 - SUCCESSFUL RUN
62 C> - COND = 1 - GRIB MESSAGE NOT FOUND
63 C> - COND = 2 - INCORRECT ARGUMENTS
64 C> - COND = 8 - ERROR ACCESSING FILE
65 C>
66  PROGRAM grb2index
67  parameter(msk1=32000,msk2=4000)

80 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

88  CALL getarg(1,cgb)

91  !CALL BASETO(1,1)
92  IF(ios.NE.0) THEN
93  lcarg=len('grb2index: Error accessing file '//cgb(1:ncgb))
94  carg(1:lcarg)='grb2index: Error accessing file '//cgb(1:ncgb)
95  CALL errmsg(carg(1:lcarg))
96  CALL errexit(8)
97  ENDIF
98  CALL getarg(2,cgi)
99  ncgi=len_trim(cgi)
100  CALL baopen(31,cgi(1:ncgi),ios)
101  IF(ios.NE.0) THEN
102  lcarg=len('grb2index: Error accessing file '//cgi(1:ncgi))
103  carg(1:lcarg)='grb2index: Error accessing file '//cgi(1:ncgi)
104  CALL errmsg(carg(1:lcarg))
105  CALL errexit(8)
106  ENDIF
107 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
108 C WRITE INDEX FILE
109  mnum=0
110  CALL getg2ir(11,msk1,msk2,mnum,cbuf,nlen,nnum,nmess,irgi)
111  IF(irgi.GT.1.OR.nnum.EQ.0.OR.nlen.EQ.0) THEN
112  CALL errmsg('grb2index: No GRIB messages detected in file '
113  & //cgb(1:ncgb))
114  CALL baclose(11,iret)
115  CALL baclose(31,iret)
116  CALL errexit(1)
117  ENDIF
118  numtot=numtot+nnum
119  mnum=mnum+nmess
120  CALL wrgi1h(31,nlen,numtot,cgb(1:ncgb))
121  iw=162
122  CALL bawrite(31,iw,nlen,kw,cbuf)
123  iw=iw+nlen
124 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
125 C EXTEND INDEX FILE IF INDEX BUFFER LENGTH TOO LARGE TO HOLD IN MEMORY
126  IF(irgi.EQ.1) THEN
127  dowhile(irgi.EQ.1.AND.nnum.GT.0)
128  IF (ASSOCIATED(cbuf)) THEN
129  DEALLOCATE(cbuf)
130  NULLIFY(cbuf)
131  ENDIF
132  CALL getg2ir(11,msk1,msk2,mnum,cbuf,nlen,nnum,nmess,irgi)
133  IF(irgi.LE.1.AND.nnum.GT.0) THEN
134  numtot=numtot+nnum
135  mnum=mnum+nmess
136  CALL bawrite(31,iw,nlen,kw,cbuf)
137  iw=iw+nlen
138  ENDIF
139  ENDDO
140  CALL wrgi1h(31,iw,numtot,cgb(1:ncgb))
141  ENDIF
142  CALL baclose(11,iret)
143  CALL baclose(31,iret)
144 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
145  END
