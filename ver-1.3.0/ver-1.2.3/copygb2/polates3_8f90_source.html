110  SUBROUTINE polates3(IPOPT,KGDSI,KGDSO,MI,MO,KM,IBI,LI,GI, &
111  NO,RLAT,RLON,IBO,LO,GO,IRET)
112  USE gdswzd_mod
113 
114  IMPLICIT NONE
115 !
116  INTEGER, INTENT(IN ) :: IBI(KM), IPOPT(20), KGDSI(200)
117  INTEGER, INTENT(IN ) :: KM, MI, MO
118  INTEGER, INTENT(INOUT) :: KGDSO(200)
119  INTEGER, INTENT( OUT) :: IBO(KM), IRET, NO
120 !
121  LOGICAL*1, INTENT(IN ) :: LI(MI,KM)
122  LOGICAL*1, INTENT( OUT) :: LO(MO,KM)
123 !
124  REAL, INTENT(IN ) :: GI(MI,KM)
125  REAL, INTENT( OUT) :: GO(MO,KM), RLAT(MO), RLON(MO)
126 !
127  REAL, PARAMETER :: FILL=-9999.
128 !cdz+1
129  REAL, PARAMETER :: FILL9=9.999000260554009e+020
130 !
131  INTEGER :: IJKGDS1, I1, J1, I2, J2, IB, JB
132  INTEGER :: IJKGDSA(20), IX, JX, IXS, JXS
133  INTEGER :: K, KXS, KXT
134  INTEGER :: LB, LSW, MP, MSPIRAL, MX
135  INTEGER :: N, NB, NB1, NB2, NB3, NB4, NV, NX
136  INTEGER :: N11(MO),N21(MO),N12(MO),N22(MO)
137 !
138  REAL :: GB, LAT(1), LON(1)
139  REAL :: PMP, RB2, RLOB(MO), RLAB(MO), WB
140  REAL :: W11(MO), W21(MO), W12(MO), W22(MO)
141  REAL :: WO(MO,KM), XF, YF, XI, YI, XX, YY
142  REAL :: XPTS(MO),YPTS(MO),XPTB(MO),YPTB(MO)
143  REAL :: XXX(1), YYY(1)
144 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
145 ! COMPUTE NUMBER OF OUTPUT POINTS AND THEIR LATITUDES AND LONGITUDES.
146 ! DO SUBSECTION OF GRID IF KGDSO(1) IS SUBTRACTED FROM 255.
147  iret=0
148  IF(kgdso(1).GE.0) THEN
149  CALL gdswzd(kgdso, 0,mo,fill,xpts,ypts,rlon,rlat,no)
150  IF(no.EQ.0) iret=3
151  ELSE
152  kgdso(1)=255+kgdso(1)
153  CALL gdswzd(kgdso,-1,mo,fill,xpts,ypts,rlon,rlat,no)
154  IF(no.EQ.0) iret=3
155  ENDIF
156 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
157 ! SET PARAMETERS
158  IF(ipopt(1).GT.16) iret=32
159  mspiral=max(ipopt(20),1)
160  nb1=ipopt(1)
161  IF(nb1.EQ.-1) nb1=2
162  IF(iret.EQ.0.AND.nb1.LT.0) iret=32
163  lsw=1
164  IF(ipopt(2).EQ.-2) lsw=2
165  IF(ipopt(1).EQ.-1.OR.ipopt(2).EQ.-1) lsw=0
166  IF(iret.EQ.0.AND.lsw.EQ.1.AND.nb1.GT.15) iret=32
167  mp=ipopt(3+ipopt(1))
168  IF(mp.EQ.-1.OR.mp.EQ.0) mp=50
169  IF(mp.LT.0.OR.mp.GT.100) iret=32
170  pmp=mp*0.01
171  IF(iret.EQ.0) THEN
172  nb2=2*nb1+1
173  rb2=1./nb2
174  nb3=nb2*nb2
175  nb4=nb3
176  IF(lsw.EQ.2) THEN
177  rb2=1./(nb1+1)
178  nb4=(nb1+1)**4
179  ELSEIF(lsw.EQ.1) THEN
180  nb4=ipopt(2)
181  DO ib=1,nb1
182  nb4=nb4+8*ib*ipopt(2+ib)
183  ENDDO
184  ENDIF
185  ELSE
186  nb3=0
187  nb4=1
188  ENDIF
189  DO k=1,km
190  DO n=1,no
191  go(n,k)=0.
192  wo(n,k)=0.
193  ENDDO
194  ENDDO
195 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
196 ! LOOP OVER SAMPLE POINTS IN OUTPUT GRID BOX
197  CALL ijkgds0(kgdsi,ijkgdsa)
198  DO nb=1,nb3
199 ! LOCATE INPUT POINTS AND COMPUTE THEIR WEIGHTS
200  jb=(nb-1)/nb2-nb1
201  ib=nb-(jb+nb1)*nb2-nb1-1
202  lb=max(abs(ib),abs(jb))
203  wb=1
204  IF(lsw.EQ.2) THEN
205  wb=(nb1+1-abs(ib))*(nb1+1-abs(jb))
206  ELSEIF(lsw.EQ.1) THEN
207  wb=ipopt(2+lb)
208  ENDIF
209  IF(wb.NE.0) THEN
210  DO n=1,no
211  xptb(n)=xpts(n)+ib*rb2
212  yptb(n)=ypts(n)+jb*rb2
213  ENDDO
214  CALL gdswzd(kgdso, 1,no,fill,xptb,yptb,rlob,rlab,nv)
215  CALL gdswzd(kgdsi,-1,no,fill,xptb,yptb,rlob,rlab,nv)
216  IF(iret.EQ.0.AND.nv.EQ.0.AND.lb.EQ.0) iret=2
217  DO n=1,no
218  xi=xptb(n)
219  yi=yptb(n)
220  IF(xi.NE.fill.AND.yi.NE.fill) THEN
221  i1=xi
222  i2=i1+1
223  j1=yi
224  j2=j1+1
225  xf=xi-i1
226  yf=yi-j1
227  n11(n)=ijkgds1(i1,j1,ijkgdsa)
228  n21(n)=ijkgds1(i2,j1,ijkgdsa)
229  n12(n)=ijkgds1(i1,j2,ijkgdsa)
230  n22(n)=ijkgds1(i2,j2,ijkgdsa)
231  IF(min(n11(n),n21(n),n12(n),n22(n)).GT.0) THEN
232  w11(n)=(1-xf)*(1-yf)
233  w21(n)=xf*(1-yf)
234  w12(n)=(1-xf)*yf
235  w22(n)=xf*yf
236  ELSE
237  n11(n)=0
238  n21(n)=0
239  n12(n)=0
240  n22(n)=0
241  ENDIF
242  ELSE
243  n11(n)=0
244  n21(n)=0
245  n12(n)=0
246  n22(n)=0
247  ENDIF
248  ENDDO
249 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
250 ! INTERPOLATE WITH OR WITHOUT BITMAPS
251  DO k=1,km
252  DO n=1,no
253  IF(n11(n).GT.0) THEN
254  IF(ibi(k).EQ.0) THEN
255 !cdz+5
256  IF(gi(n11(n),k) == fill9 .OR. gi(n21(n),k) == fill9 .OR. &
257  gi(n12(n),k) == fill9 .OR. gi(n22(n),k) == fill9) THEN
258  go(n,k)=fill9
259  lo(n,k)=.false.
260  ELSE
261  gb=w11(n)*gi(n11(n),k)+w21(n)*gi(n21(n),k) &
262  +w12(n)*gi(n12(n),k)+w22(n)*gi(n22(n),k)
263  go(n,k)=go(n,k)+wb*gb
264  wo(n,k)=wo(n,k)+wb
265 !cdz+1
266  ENDIF
267  ELSE
268  IF(li(n11(n),k)) THEN
269 !cdz+4
270  IF(gi(n11(n),k) == fill9) THEN
271  go(n,k)=fill9
272  lo(n,k)=.false.
273  ELSE
274  go(n,k)=go(n,k)+wb*w11(n)*gi(n11(n),k)
275  wo(n,k)=wo(n,k)+wb*w11(n)
276 !cdz+1
277  ENDIF
278  ENDIF
279  IF(li(n21(n),k)) THEN
280 !cdz+4
281  IF(gi(n21(n),k) == fill9) THEN
282  go(n,k)=fill9
283  lo(n,k)=.false.
284  ELSE
285  go(n,k)=go(n,k)+wb*w21(n)*gi(n21(n),k)
286  wo(n,k)=wo(n,k)+wb*w21(n)
287 !cdz+1
288  ENDIF
289  ENDIF
290  IF(li(n12(n),k)) THEN
291 !cdz+4
292  IF(gi(n12(n),k) == fill9) THEN
293  go(n,k)=fill9
294  lo(n,k)=.false.
295  ELSE
296  go(n,k)=go(n,k)+wb*w12(n)*gi(n12(n),k)
297  wo(n,k)=wo(n,k)+wb*w12(n)
298 !cdz+1
299  ENDIF
300  ENDIF
301  IF(li(n22(n),k)) THEN
302 !cdz+4
303  IF(gi(n22(n),k) == fill9) THEN
304  go(n,k)=fill9
305  lo(n,k)=.false.
306  ELSE
307  go(n,k)=go(n,k)+wb*w22(n)*gi(n22(n),k)
308  wo(n,k)=wo(n,k)+wb*w22(n)
309 !cdz+1
310  ENDIF
311  ENDIF
312  ENDIF
313  ENDIF
314  ENDDO
315  ENDDO
316  ENDIF
317  ENDDO ! sub-grid points
318 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
319 ! COMPUTE OUTPUT BITMAPS AND FIELDS
320  km_loop : DO k=1,km
321  ibo(k)=ibi(k)
322  n_loop : DO n=1,no
323  lo(n,k)=wo(n,k).GE.pmp*nb4
324  IF(lo(n,k)) THEN
325  go(n,k)=go(n,k)/wo(n,k)
326  ELSEIF (mspiral.GT.1) THEN
327  lat(1)=rlat(n)
328  lon(1)=rlon(n)
329  CALL gdswzd(kgdsi,-1,1,fill,xxx,yyy,lon,lat,nv)
330  xx=xxx(1)
331  yy=yyy(1)
332  IF(nv.EQ.1)THEN
333  i1=nint(xx)
334  j1=nint(yy)
335  ixs=sign(1.,xx-i1)
336  jxs=sign(1.,yy-j1)
337  spiral_loop : DO mx=2,mspiral**2
338  kxs=sqrt(4*mx-2.5)
339  kxt=mx-(kxs**2/4+1)
340  SELECT CASE(mod(kxs,4))
341  CASE(1)
342  ix=i1-ixs*(kxs/4-kxt)
343  jx=j1-jxs*kxs/4
344  CASE(2)
345  ix=i1+ixs*(1+kxs/4)
346  jx=j1-jxs*(kxs/4-kxt)
347  CASE(3)
348  ix=i1+ixs*(1+kxs/4-kxt)
349  jx=j1+jxs*(1+kxs/4)
350  CASE DEFAULT
351  ix=i1-ixs*kxs/4
352  jx=j1+jxs*(kxs/4-kxt)
353  END SELECT
354  nx=ijkgds1(ix,jx,ijkgdsa)
355  IF(nx.GT.0.)THEN
356  IF(li(nx,k).OR.ibi(k).EQ.0) THEN
357  go(n,k)=gi(nx,k)
358  lo(n,k)=.true.
359 !cdz+1
360  IF(gi(nx,k) == fill9) lo(n,k)=.false.
361  cycle n_loop
362  ENDIF
363  ENDIF
364  ENDDO spiral_loop
365  ibo(k)=1
366  go(n,k)=0.
367  ELSE
368  ibo(k)=1
369  go(n,k)=0.
370  ENDIF
371  ELSE ! no spiral search option
372  ibo(k)=1
373  go(n,k)=0.
374  ENDIF
375  ENDDO n_loop
376  ENDDO km_loop
377  IF(kgdso(1).EQ.0) CALL polfixs(no,mo,km,rlat,rlon,ibo,lo,go)
378 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
379  END SUBROUTINE polates3
